def matxMultiply(A, B):
    row_A,col_A = shape(A)
    row_B,col_B = shape(B)
    if(col_A != row_B):
        raise ValueError("can not matxMultiply")
    C = []
    for i in range(row_A):
        C.append([])
        for j in range(col_B):
            
            sum_of_mutiply = 0
            for k in range(col_A):
                 sum_of_mutiply += A[i][k] * B[k][j]
            C[i].append(sum_of_mutiply)
    return C

def swap_max_abs_row(A,index):
    r,c = shape(A)
    max_index = index
    max_abs = abs(A[index][index])
    for k in range(index+1,r):
        if(abs(A[k][index])>max_abs):
            max_index,max_abs = k,abs(A[k][index])
           
    if(index != max_index):
        swapRows(A,index,max_index)

def gj_Solve(A, b, decPts=4, epsilon=1.0e-16):
    print_matrix(A)
    print_matrix(b)

    Ab = augmentMatrix(A,b)

    r,c = shape(Ab)
    for i in range(r):
        swap_max_abs_row(Ab,i)
        if(abs(Ab[i][i])<=epsilon):
            return None
        scaleRow(Ab,i,1.0/Ab[i][i])
        
        for j in range(r):

                scale = -Ab[j][i]/Ab[i][i]
                addScaledRow(Ab,j,i,scale)
      
        print_matrix(Ab)
    matxRound(Ab,decPts)
    R = []
    for i in range(r):
        R.append([])
        R[i] .append(Ab[i][r])
        
    return R
